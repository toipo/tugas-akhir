class _PresentaseRomCerdas1State extends State<PresentaseRomCerdas1> {
  late DatabaseReference _dbRef;
  StreamSubscription? _dbSubscription;
  String _nilaiTeks = "...";
  bool _isLiveMode = false;

  @override
  void initState() {
    super.initState();
    _checkLiveMode();
    _initSequence();
  }

  @override
  void didUpdateWidget(PresentaseRomCerdas1 oldWidget) {
    super.didUpdateWidget(oldWidget);
    if (oldWidget.dateKey != widget.dateKey ||
        oldWidget.dbPath != widget.dbPath) {
      _dbSubscription?.cancel();
      setState(() {
        _nilaiTeks = "...";
      });
      _checkLiveMode();
      _initSequence();
    }
  }

  void _checkLiveMode() {
    DateTime now = DateTime.now();
    String todayStr =
        "${now.year}-${now.month.toString().padLeft(2, '0')}-${now.day.toString().padLeft(2, '0')}";
    _isLiveMode = (widget.dateKey == todayStr);
  }

  Future<void> _initSequence() async {
    _setupFirebaseRefs();
    await _loadLocalCache();
    if (_isLiveMode) {
      Future.delayed(const Duration(milliseconds: 200), () {
        if (mounted) _listenToFirebase();
      });
    } else {
      if (_nilaiTeks == "..." || _nilaiTeks == "0%") {
        await _getHistoryFromFirebaseOnce();
      }
    }
  }

  void _setupFirebaseRefs() {
    FirebaseDatabase database = FirebaseDatabase.instanceFor(
        app: Firebase.app(),
        databaseURL:
            'https://project-ta-9d176-default-rtdb.asia-southeast1.firebasedatabase.app');
    String finalPath = widget.dbPath;
    if (!finalPath.endsWith(widget.dateKey)) {
      finalPath = "$finalPath/${widget.dateKey}";
    }
    // PERUBAHAN: Target pembacaan database diubah menjadi presentase_rom
    _dbRef = database.ref('$finalPath/presentase_rom');
  }

  Future<void> _loadLocalCache() async {
    SharedPreferences prefs = await SharedPreferences.getInstance();
    // PERUBAHAN: Key cache diubah menjadi presentase_rom
    String? cachedData = prefs.getString('presentase_rom_1_${widget.dateKey}');
    if (mounted) {
      setState(() {
        if (cachedData != null) {
          _nilaiTeks = cachedData;
        } else if (_isLiveMode) {
          _nilaiTeks = "0%";
        }
      });
    }
  }

  Future<void> _getHistoryFromFirebaseOnce() async {
    try {
      DataSnapshot snapshot =
          await _dbRef.get().timeout(const Duration(seconds: 10));
      _processSnapshot(snapshot);
    } catch (e) {
      if (mounted) {
        setState(() {
          _nilaiTeks = "0%";
        });
      }
    }
  }

  void _listenToFirebase() {
    _dbSubscription = _dbRef.onValue.listen((event) async {
      _processSnapshot(event.snapshot);
    });
  }

  Future<void> _processSnapshot(DataSnapshot snapshot) async {
    SharedPreferences prefs = await SharedPreferences.getInstance();
    if (snapshot.exists && snapshot.value != null) {
      double val = double.tryParse(snapshot.value.toString()) ?? 0.0;
      String result = '${val.toStringAsFixed(1)}%';
      if (mounted) {
        setState(() {
          _nilaiTeks = result;
        });
      }
      // PERUBAHAN: Key cache diubah menjadi presentase_rom
      prefs.setString('presentase_rom_1_${widget.dateKey}', result);
    } else {
      if (mounted) {
        setState(() {
          _nilaiTeks = "0%";
        });
      }
      // PERUBAHAN: Key cache diubah menjadi presentase_rom
      prefs.remove('presentase_rom_1_${widget.dateKey}');
    }
  }

  @override
  void dispose() {
    _dbSubscription?.cancel();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Container(
      width: widget.width,
      height: widget.height,
      alignment: Alignment.center,
      child: Text(
        _nilaiTeks,
        style: TextStyle(
          fontSize: widget.textSize,
          color: widget.textColor,
          fontWeight: FontWeight.bold,
        ),
      ),
    );
  }
}
