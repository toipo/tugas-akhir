    SharedPreferences pr = await SharedPreferences.getInstance();

  void _processData(dynamic val, SharedPreferences pr) {
    if (val == null) {
      if (mounted) setState(() => _nilaiTeks = "0");
      pr.setString('cache_rom_8_${widget.dateKey}', "0");
      return;

    else if (val is List) {

                    0.0)

      }
    });

  }
