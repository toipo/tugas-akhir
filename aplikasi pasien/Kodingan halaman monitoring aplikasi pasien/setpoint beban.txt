Future bacaSetpointRTDB1(
  BuildContext context,
  String dateKey,
  String idPasien, // 🔥 PARAMETER BARU
) async {
  // 🔥 SANITASI TANGGAL (Biar format pasti YYYY-MM-DD seperti widget lainnya)
  String cleanDateKey = dateKey;
  if (cleanDateKey.length > 10 && cleanDateKey.contains(' ')) {
    cleanDateKey = cleanDateKey.split(' ')[0];
  }
  List<String> dateParts = cleanDateKey.split('-');
  if (dateParts.length == 3 && dateParts[0].length == 2) {
    cleanDateKey = "${dateParts[2]}-${dateParts[1]}-${dateParts[0]}";
  }

  // 1. CEK APAKAH DATEKEY ADALAH HARI INI
  DateTime now = DateTime.now();
  String todayStr =
      "${now.year}-${now.month.toString().padLeft(2, '0')}-${now.day.toString().padLeft(2, '0')}";

  bool isLiveMode = (cleanDateKey == todayStr);

  // 2. BACA DARI MEMORI HP DULU
  SharedPreferences prefs = await SharedPreferences.getInstance();
  String? cachedMin = prefs.getString('setpoint_min_$cleanDateKey');
  String? cachedMax = prefs.getString('setpoint_max_$cleanDateKey');

  // Jika di HP ada data riwayat, tampilkan langsung biar cepat
  if (cachedMin != null && cachedMax != null) {
    FFAppState().update(() {
      FFAppState().setpointBerat = "$cachedMin% - $cachedMax%";
    });
  }

  // SETUP FIREBASE
  String myDatabaseUrl =
      'https://project-ta-9d176-default-rtdb.asia-southeast1.firebasedatabase.app';
  FirebaseDatabase database = FirebaseDatabase.instanceFor(
    app: Firebase.app(),
    databaseURL: myDatabaseUrl,
  );

  // 3. LOGIKA PEMBACAAN BERDASARKAN MODE (LIVE vs HISTORY)
  if (isLiveMode) {
    // 🔥 MODE HARI INI: Pantau terus dari folder '/pengaturan'
    DatabaseReference dbRef = database.ref('$idPasien/pengaturan');

    dbRef.onValue.listen((event) async {
      if (event.snapshot.value != null) {
        Map<dynamic, dynamic> data =
            event.snapshot.value as Map<dynamic, dynamic>;

        String min = data['setpoint_min']?.toString() ?? "0";
        String max = data['setpoint_max']?.toString() ?? "0";

        FFAppState().update(() {
          FFAppState().setpointBerat = "$min - $max";
        });

        // Simpan ke memori HP
        prefs.setString('setpoint_min_$cleanDateKey', min);
        prefs.setString('setpoint_max_$cleanDateKey', max);
      }
    });
  } else {
    // 🔥 MODE RIWAYAT (MASA LALU): Ambil dari folder '/riwayat/Tanggal' jika di HP kosong/belum sinkron
    if (cachedMin == null || cachedMax == null) {
      DatabaseReference riwayatRef =
          database.ref('$idPasien/riwayat/$cleanDateKey');
      DatabaseEvent event = await riwayatRef.once();

      if (event.snapshot.value != null) {
        Map<dynamic, dynamic> dataHariIni =
            event.snapshot.value as Map<dynamic, dynamic>;

        // 🔥 PERUBAHAN DI SINI: Langsung ambil datanya tanpa perlu looping forEach
        String min = dataHariIni['setpoint_min']?.toString() ?? "0";
        String max = dataHariIni['setpoint_max']?.toString() ?? "0";

        // Tampilkan ke layar
        FFAppState().update(() {
          FFAppState().setpointBerat = "$min - $max";
        });

        // Simpan permanen ke internal HP biar besok-besok nggak perlu download lagi
        prefs.setString('setpoint_min_$cleanDateKey', min);
        prefs.setString('setpoint_max_$cleanDateKey', max);

        print(
            "Berhasil narik riwayat setpoint tanggal $cleanDateKey: $min - $max");
      } else {
        // Kalau di hari itu emang alatnya nggak dinyalain (data kosong)
        FFAppState().update(() {
          FFAppState().setpointBerat = "0% - 0%";
        });
      }
    }
  }
}
