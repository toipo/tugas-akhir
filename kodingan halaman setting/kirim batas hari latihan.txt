Future simpanBatasHari(
  BuildContext context,
  String? batasHari,
  String dbPath, // 🔥 Parameter dbPath ditambahkan di sini
) async {
  // 1. Validasi Kosong
  if (batasHari == null || batasHari.trim().isEmpty) {
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
          content: Text('Batas hari latihan tidak boleh kosong!'),
          backgroundColor: Colors.red),
    );
    return;
  }

  // 2. Validasi Harus Angka
  if (int.tryParse(batasHari.trim()) == null) {
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
          content: Text('Batas hari harus berupa angka!'),
          backgroundColor: Colors.red),
    );
    return;
  }

  // 3. Kirim ke Firebase RTDB menggunakan parameter dbPath
  // Pastikan saat memanggil action ini, dbPath diisi "pasien_1/pengaturan"
  final String dbUrl =
      'https://project-ta-9d176-default-rtdb.asia-southeast1.firebasedatabase.app/$dbPath.json';

  try {
    final response = await http.patch(
      Uri.parse(dbUrl),
      body: json.encode({'batas_hari': int.parse(batasHari.trim())}),
    );

    if (response.statusCode == 200) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(
            content: Text('Batas Hari Latihan berhasil disimpan!'),
            backgroundColor: Colors.green),
      );
    } else {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(
            content: Text('Gagal menyimpan ke database!'),
            backgroundColor: Colors.red),
      );
    }
  } catch (e) {
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
          content: Text('Terjadi kesalahan koneksi!'),
          backgroundColor: Colors.red),
    );
  }
}
