Future simpanSetpointBerat(
  BuildContext context,
  String? minVal,
  String? maxVal,
  String dbPath, // 🔥 Parameter baru ditambahkan di sini
) async {
  // 1. Validasi Kosong
  if (minVal == null ||
      minVal.trim().isEmpty ||
      maxVal == null ||
      maxVal.trim().isEmpty) {
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
          content: Text('Form setpoint tidak boleh ada yang kosong!'),
          backgroundColor: Colors.red),
    );
    return;
  }

  // 2. Validasi Tanda Persen (%)
  if (!minVal.contains('%') || !maxVal.contains('%')) {
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
          content: Text('Nilai setpoint berat harus menggunakan tanda %'),
          backgroundColor: Colors.red),
    );
    return;
  }

  // 3. Kirim ke Firebase RTDB
  // 🔥 URL sekarang menggunakan parameter dbPath.
  // Pastikan parameter dbPath yang dikirim berisi "pasien_1/pengaturan"
  final String dbUrl =
      'https://project-ta-9d176-default-rtdb.asia-southeast1.firebasedatabase.app/$dbPath.json';

  try {
    final response = await http.patch(
      Uri.parse(dbUrl),
      body: json.encode(
          {'setpoint_min': minVal.trim(), 'setpoint_max': maxVal.trim()}),
    );

    if (response.statusCode == 200) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(
            content: Text('Setpoint Berat berhasil disimpan!'),
            backgroundColor: Colors.green),
      );
    } else {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(
            content: Text('Gagal menyimpan ke database!'),
            backgroundColor: Colors.red),
      );
    }
  } catch (e) {
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
          content: Text('Terjadi kesalahan koneksi!'),
          backgroundColor: Colors.red),
    );
  }
