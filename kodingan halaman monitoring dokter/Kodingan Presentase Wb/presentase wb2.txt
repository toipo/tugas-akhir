class _PresentaseWbWidget2State extends State<PresentaseWbWidget2> {
  late FirebaseDatabase _database;
  StreamSubscription<DatabaseEvent>? _monitoringSub;
  StreamSubscription<DatabaseEvent>? _settingsSub;
  Timer? _dayChecker;

  double _presentase = 0.0;
  bool _isLoading = true;
  String _currentDateStr = '';
  double _minPercent = 0.0;
  double _maxPercent = 100.0;

  @override
  void initState() {
    super.initState();
    _initDatabase();
    _startListening();
    _startDayChecker();
  }

  void _initDatabase() {
    String myDatabaseUrl =
        'https://project-ta-9d176-default-rtdb.asia-southeast1.firebasedatabase.app';
    _database = FirebaseDatabase.instanceFor(
        app: Firebase.app(), databaseURL: myDatabaseUrl);
  }

  String _getTodayString() {
    DateTime now = DateTime.now();
    return "${now.year}-${now.month.toString().padLeft(2, '0')}-${now.day.toString().padLeft(2, '0')}";
  }

  void _startListening() {
    _currentDateStr = _getTodayString();
    String cleanBasePath = widget.basePath.endsWith('/')
        ? widget.basePath.substring(0, widget.basePath.length - 1)
        : widget.basePath;

    DatabaseReference monitoringRef =
        _database.ref('$cleanBasePath/monitoring/$_currentDateStr');
    DatabaseReference settingsRef = _database.ref('$cleanBasePath/pengaturan');
    DatabaseReference riwayatRef =
        _database.ref('$cleanBasePath/riwayat/$_currentDateStr/presentase_wb');

    _monitoringSub?.cancel();
    _settingsSub?.cancel();

    _settingsSub = settingsRef.onValue.listen((event) {
      if (event.snapshot.exists &&
          event.snapshot.value != null &&
          event.snapshot.value is Map) {
        Map data = event.snapshot.value as Map;
        String minStr = data['setpoint_min']
                ?.toString()
                .replaceAll(RegExp(r'[^0-9.]'), '') ??
            "0";
        String maxStr = data['setpoint_max']
                ?.toString()
                .replaceAll(RegExp(r'[^0-9.]'), '') ??
            "100";
        if (mounted) {
          setState(() {
            _minPercent = double.tryParse(minStr) ?? 0.0;
            _maxPercent = double.tryParse(maxStr) ?? 100.0;
          });
        }
      }
    });

    _monitoringSub = monitoringRef.onValue.listen((event) {
      if (event.snapshot.exists &&
          event.snapshot.value != null &&
          event.snapshot.value is Map) {
        Map data = event.snapshot.value as Map;
        int totalData = 0, validData = 0;

        data.forEach((key, value) {
          if (key.toString().toLowerCase().contains('datawb')) {
            String cleanVal =
                value.toString().replaceAll(RegExp(r'[^0-9.]'), '');
            if (cleanVal.isNotEmpty) {
              totalData++;
              double parsed = double.tryParse(cleanVal) ?? 0.0;
              double pct = parsed > 1.0 ? parsed : parsed * 100;
              if (pct >= _minPercent && pct <= _maxPercent) validData++;
            }
          }
        });

        double finalPercentage =
            totalData > 0 ? (validData / totalData) * 100 : 0.0;
        if (mounted) {
          setState(() {
            _presentase = finalPercentage;
            _isLoading = false;
          });
        }
        riwayatRef.set(finalPercentage.toStringAsFixed(1));
      } else {
        if (mounted)
          setState(() {
            _presentase = 0.0;
            _isLoading = false;
          });
        riwayatRef.set("0.0");
      }
    }, onError: (_) {
      if (mounted)
        setState(() {
          _presentase = 0.0;
          _isLoading = false;
        });
    });
  }

  void _startDayChecker() {
    _dayChecker = Timer.periodic(const Duration(minutes: 1), (timer) {
      if (_getTodayString() != _currentDateStr) {
        if (mounted)
          setState(() {
            _presentase = 0.0;
            _isLoading = true;
          });
        _startListening();
      }
    });
  }

  @override
  void dispose() {
    _monitoringSub?.cancel();
    _settingsSub?.cancel();
    _dayChecker?.cancel();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Container(
      width: widget.width,
      height: widget.height,
      alignment: Alignment.center,
      child: _isLoading
          ? const SizedBox(
              width: 24,
              height: 24,
              child: CircularProgressIndicator(strokeWidth: 2))
          : Text('${_presentase.toStringAsFixed(1)}%',
              style: TextStyle(
                  fontSize: widget.textSize ?? 20.0,
                  fontWeight: FontWeight.bold,
                  color: widget.textColor ?? Colors.black)),
    );
  }
}
