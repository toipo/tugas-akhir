class PresentaseRomWidget1 extends StatefulWidget {
  const PresentaseRomWidget1({
    super.key,
    this.width,
    this.height,
    required this.basePath,
    this.textColor,
    this.textSize,
  });

  final double? width;
  final double? height;
  final String basePath;
  final Color? textColor;
  final double? textSize;

  @override
  State<PresentaseRomWidget1> createState() => _PresentaseRomWidget1State();
}

class _PresentaseRomWidget1State extends State<PresentaseRomWidget1> {
  late FirebaseDatabase _database;
  StreamSubscription<DatabaseEvent>? _monitoringSub;
  Timer? _dayChecker;

  double _presentase = 0.0;
  bool _isLoading = true;
  String _currentDateStr = '';

  @override
  void initState() {
    super.initState();
    _initDatabase();
    _startListening();
    _startDayChecker();
  }

  void _initDatabase() {
    String myDatabaseUrl =
        'https://project-ta-9d176-default-rtdb.asia-southeast1.firebasedatabase.app';
    _database = FirebaseDatabase.instanceFor(
        app: Firebase.app(), databaseURL: myDatabaseUrl);
  }

  String _getTodayString() {
    DateTime now = DateTime.now();
    return "${now.year}-${now.month.toString().padLeft(2, '0')}-${now.day.toString().padLeft(2, '0')}";
  }

  void _startListening() {
    _currentDateStr = _getTodayString();

    // Membersihkan basePath jika ada garis miring berlebih di akhir
    String cleanBasePath = widget.basePath.endsWith('/')
        ? widget.basePath.substring(0, widget.basePath.length - 1)
        : widget.basePath;

    // Path untuk membaca monitoring data ROM
    DatabaseReference monitoringRef =
        _database.ref('$cleanBasePath/monitoring/$_currentDateStr');

    // Path untuk menyimpan hasil persentase ROM
    DatabaseReference riwayatRef =
        _database.ref('$cleanBasePath/riwayat/$_currentDateStr/presentase_rom');

    _monitoringSub?.cancel();

    _monitoringSub = monitoringRef.onValue.listen((event) {
      if (event.snapshot.exists &&
          event.snapshot.value != null &&
          event.snapshot.value is Map) {
        Map data = event.snapshot.value as Map;
        int totalData = 0;
        int validData = 0;

        // Mencari semua data yang namanya mengandung 'datarom'
        data.forEach((key, value) {
          if (key.toString().toLowerCase().contains('datarom')) {
            totalData++;

            // Mengecek apakah nilainya true.
            // Kita ubah ke string huruf kecil semua untuk jaga-jaga format dari Firebase
            String strValue = value.toString().toLowerCase().trim();
            if (strValue == 'true' || strValue == '1') {
              validData++;
            }
          }
        });

        // Hitung persentase: (Valid / Total) * 100
        double finalPercentage =
            totalData > 0 ? (validData / totalData) * 100 : 0.0;

        if (mounted) {
          setState(() {
            _presentase = finalPercentage;
            _isLoading = false;
          });
        }

        // Simpan hasil persentase ROM ke Firebase Riwayat
        riwayatRef.set(finalPercentage.toStringAsFixed(1));
      } else {
        if (mounted) {
          setState(() {
            _presentase = 0.0;
            _isLoading = false;
          });
        }
        riwayatRef.set("0.0");
      }
    }, onError: (_) {
      if (mounted) {
        setState(() {
          _presentase = 0.0;
          _isLoading = false;
        });
      }
    });
  }

  void _startDayChecker() {
    _dayChecker = Timer.periodic(const Duration(minutes: 1), (timer) {
      if (_getTodayString() != _currentDateStr) {
        if (mounted) {
          setState(() {
            _presentase = 0.0;
            _isLoading = true;
          });
        }
        _startListening();
      }
    });
  }

  @override
  void dispose() {
    _monitoringSub?.cancel();
    _dayChecker?.cancel();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Container(
      width: widget.width,
      height: widget.height,
      alignment: Alignment.center,
      child: _isLoading
          ? const SizedBox(
              width: 24,
              height: 24,
              child: CircularProgressIndicator(strokeWidth: 2))
          : Text('${_presentase.toStringAsFixed(1)}%',
              style: TextStyle(
                  fontSize: widget.textSize ?? 20.0,
                  fontWeight: FontWeight.bold,
                  color: widget.textColor ?? Colors.black)),
    );
  }
}
