class RealtimeCounter2 extends StatefulWidget {
  const RealtimeCounter2({
    super.key,
    this.width,
    this.height,
    required this.dbPath,
    required this.textColor,
    required this.textSize,
  });

  final double? width;
  final double? height;
  final String dbPath;
  final Color textColor;
  final double textSize;

  @override
  State<RealtimeCounter2> createState() => _RealtimeCounterState2();
}

class _RealtimeCounterState2 extends State<RealtimeCounter2> {
  late FirebaseDatabase _database;
  late DatabaseReference _dbRef;

  StreamSubscription? _dbSubscription;
  Timer? _dayCheckerTimer;

  String _nilaiTeks = "...";
  String _currentDateStr = "";

  @override
  void initState() {
    super.initState();

    _initDayAndLoadCache();

    String myDatabaseUrl =
        'https://project-ta-9d176-default-rtdb.asia-southeast1.firebasedatabase.app';
    _database = FirebaseDatabase.instanceFor(
      app: Firebase.app(),
      databaseURL: myDatabaseUrl,
    );

    _database.setPersistenceEnabled(true);
    _dbRef = _database.ref('${widget.dbPath}/$_currentDateStr');

    _startDayCheckerTimer();
    _listenToFirebase();
  }

  Future<void> _initDayAndLoadCache() async {
    DateTime now = DateTime.now();
    _currentDateStr =
        "${now.year}-${now.month.toString().padLeft(2, '0')}-${now.day.toString().padLeft(2, '0')}";

    SharedPreferences prefs = await SharedPreferences.getInstance();
    int? cachedCounter = prefs.getInt('counter2_rom_$_currentDateStr');

    if (mounted) {
      setState(() {
        _nilaiTeks = cachedCounter?.toString() ?? "0";
      });
    }
  }

  void _startDayCheckerTimer() {
    _dayCheckerTimer = Timer.periodic(const Duration(minutes: 1), (timer) {
      DateTime now = DateTime.now();
      String todayStr =
          "${now.year}-${now.month.toString().padLeft(2, '0')}-${now.day.toString().padLeft(2, '0')}";

      if (_currentDateStr != "" && _currentDateStr != todayStr) {
        if (mounted) {
          setState(() {
            _nilaiTeks = "0";
            _currentDateStr = todayStr;
          });

          _dbRef = _database.ref('${widget.dbPath}/$_currentDateStr');
          _listenToFirebase();
        }
      }
    });
  }

  void _listenToFirebase() {
    _dbSubscription?.cancel();
    _dbSubscription = _dbRef.onValue.listen((event) async {
      if (event.snapshot.value != null && event.snapshot.value is Map) {
        Map<dynamic, dynamic> data =
            event.snapshot.value as Map<dynamic, dynamic>;

        int totalHitunganRom = 0;

        // 🔥 PERBAIKAN: Menghitung total data ROM yang bernilai true / 1
        data.forEach((key, value) {
          String keyStr = key.toString().toLowerCase();

          // Cek jika nama kuncinya mengandung 'rom' atau 'datarom'
          if (keyStr.contains('rom') || keyStr.contains('datarom')) {
            // Evaluasi apakah nilainya adalah boolean true, string 'true', atau angka 1
            if (value == true ||
                value.toString().toLowerCase() == 'true' ||
                value.toString() == '1') {
              totalHitunganRom++; // Tambahkan 1 ke counter
            }
          }
        });

        // Tampilkan hasil perhitungan ke teks UI
        if (mounted) {
          setState(() {
            _nilaiTeks = totalHitunganRom.toString();
          });
        }

        // Simpan hasil hitungan ke penyimpanan lokal HP (Cache)
        SharedPreferences prefs = await SharedPreferences.getInstance();
        prefs.setInt('counter2_rom_$_currentDateStr', totalHitunganRom);

        // 🔥 PERBAIKAN: Kirim hasil hitungan akhir ke node riwayat
        try {
          String riwayatPathBase =
              widget.dbPath.replaceAll('monitoring', 'riwayat');
          DatabaseReference riwayatRef =
              _database.ref('$riwayatPathBase/$_currentDateStr');

          // Disimpan sebagai 'total_rom' agar terbaca oleh custom function riwayat Anda
          await riwayatRef.update({
            'total_rom': totalHitunganRom,
          });
        } catch (e) {
          debugPrint("Gagal sinkron riwayat ROM: $e");
        }
      } else {
        if (mounted) {
          setState(() {
            _nilaiTeks = "0";
          });
        }
        SharedPreferences prefs = await SharedPreferences.getInstance();
        prefs.setInt('counter2_rom_$_currentDateStr', 0);
      }
    });
  }

  @override
  void dispose() {
    _dbSubscription?.cancel();
    _dayCheckerTimer?.cancel();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Container(
      width: widget.width,
      height: widget.height,
      alignment: Alignment.center,
      child: Text(
        _nilaiTeks,
        style: TextStyle(
          fontSize: widget.textSize,
          color: widget.textColor,
          fontWeight: FontWeight.bold,
        ),
      ),
    );
  }
}
